# 前缀和差分

## 一维前缀和
```cpp
int q[100];
int a[100];
int n, m;

int main(int argc, char *argv[]) {
    freopen("data", "r", stdin);
    cin >> n >> m;
    int idx = 1;
    while (idx <= n) {
        cin >> q[idx];
        a[idx] = a[idx - 1] + q[idx];
        idx++;
    }

    int l, r;
    while (cin >> l >> r) {
        cout << a[r] - a[l - 1] << endl;
    }
    return 0;
}
```
## 二维前缀和
```cpp
int p[10005];

int find(int x) {
    if (x != p[x])
        return p[x] = find(p[x]);
    else
        return x;
}

void Union(int x, int y) {
    int a = find(x), b = find(y);
    if (a == b) return;
    p[a] = b;
}

int main(int argc, char *argv[]) {
    for (int i = 1; i <= 10; i++) {
        p[i] = i;
    }
    Union(1, 2);
    Union(2, 3);

    for (int i = 1; i <= 10; i++) {
        cout << find(i) << endl;
    }

    return 0;
}
```

## 一维差分
```cpp
int n, m;
int q[10005];
int a[10005];

int main(int argc, char *argv[]) {
    freopen("data", "r", stdin);
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> q[i];
        a[i] = q[i] - q[i - 1];
    }

    while (m--) {
        int l, r, c;
        cin >> l >> r >> c;
        a[l] += c;
        a[r + 1] -= c;
    }

    for (int i = 1; i <= n; i++) {
        a[i] += a[i - 1];
        cout << a[i] << " ";
    }
    return 0;
}
```

## 二维差分
```cpp
int n, m, q;
#define SIZE 100
int a[SIZE][SIZE], num[SIZE][SIZE];

void insert(int x, int y, int c) {
    a[x][y] += c;
    a[x + 1][y] -= c;
    a[x][y + 1] -= c;
    a[x + 1][y + 1] += c;
}

int main(int argc, char *argv[]) {
    freopen("data", "r", stdin);
    cin >> n >> m >> q;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cin >> num[i][j];
            insert(i, j, num[i][j]);
            cout << a[i][j] << " ";
        }
        cout << endl;
    }
    cout << endl;
    while (q--) {
        int x1, y1, x2, y2, c;
        cin >> x1 >> y1 >> x2 >> y2 >> c;
        a[x1][y1] += c;
        a[x2 + 1][y1] -= c;
        a[x1][y2 + 1] -= c;
        a[x2 + 1][y2 + 1] += c;
    }
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            a[i][j] += a[i - 1][j] + a[i][j - 1] - a[i - 1][j - 1];
            cout << a[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
```
