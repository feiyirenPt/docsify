# 八大排序

```cpp
void bubbleSort(int q[], int n) {
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (q[j] > q[j + 1]) swap(q[j], q[j + 1]);
        }
    }
}
```
```cpp
void SelectSort(int q[], int n) {
    for (int i = 0; i < n; i++) {
        int min = i;
        for (int j = i + 1; j < n; j++) {
            if (q[min] > q[j]) {
                min = j;
            }
        }
        swap(q[min], q[i]);
    }
}
```

```cpp
void insertSort(int q[], int n) {
    for (int i = 1; i < n; i++) {
        int j = i - 1;
        int tmp = q[i];
        while (j >= 0 && q[j] > tmp) {
            q[j + 1] = q[j];
            j--;
        }
        q[j + 1] = tmp;
    }
}
```

```cpp
void insertSort2(int q[], int n) {
```

```cpp
void quickSort(int nums[], int l, int r) {
    if (l >= r) return;
    int mid = (l + r) >> 1;
    int i = l - 1, j = r + 1;
    int p = nums[mid];
    while (i < j) {
        do i++;
        while (nums[i] < p);
        do j--;
        while (nums[j] > p);
        if (i < j) swap(nums[i], nums[j]);
    }

    quickSort(nums, l, j);
    quickSort(nums, j + 1, r);
}
```

```cpp
        }

    while (i <= mid) tmp[k++] = nums[i++];
```

```cpp
void shellSort(int q[], int n) {
    for (int gap = n / 2; gap > 0; gap /= 2) {
        for (int i = gap; i < n; i++) {
            for (int j = i - gap; j >= 0 && q[j + gap] < q[j]; j -= gap) {
                swap(q[j + gap], q[j]);
            }
        }
    }
}
```

```cpp
void down(int h[], int u, int size) {
    int t = u;
    if (2 * u <= size && h[2 * u] < h[u]) {
        t = 2 * u;
    }
    if (2 * u + 1 <= size && h[2 * u + 1] < h[t]) {
        t = 2 * u + 1;
    }
    if (t != u) {
        swap(h[t], h[u]);
        down(h, t, size);
    }
}
```

```cpp
void heapSort(int q[], int size) {
    for (int i = size; i >= 0; i--) {
        q[i + 1] = q[i];
    }
    for (int i = size / 2; i; i--) {
        down(q, i, size);
    }
}

void heapPrint(int q[], int size) {
    while (size) {
        cout << q[1] << " ";
        swap(q[size], q[1]);
        size--;
        down(q, 1, size);
    }
}
```
